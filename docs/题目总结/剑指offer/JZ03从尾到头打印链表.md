# 解析

## 递归方法

```C++
/**
*  struct ListNode {
*        int val;
*        struct ListNode *next;
*        ListNode(int x) :
*              val(x), next(NULL) {
*        }
*  };
*/
class Solution {
public:
    vector<int> printListFromTailToHead(ListNode* head) {
        //递归解法
        vector<int> ans;
        // head 为空 ！head 为 true 返回 0 
        // head 非空, 执行ans=printfListFormTailToHead()递归
        if(!head) return ans; 
        ans = printListFromTailToHead(head->next);
        ans.push_back(head->val);
        return ans;
    }
};
```



## 反转链表

```C++
/**
*  struct ListNode {
*        int val;
*        struct ListNode *next;
*        ListNode(int x) :
*              val(x), next(NULL) {
*        }
*  };
*/
class Solution {
public:
    vector<int> printListFromTailToHead(ListNode* head) {
        ListNode* pre = nullptr;
        ListNode* cur = head;
        ListNode* temp = cur;
        
        while(cur)
        {
            temp = cur->next;
            cur->next = pre;//后面步骤将pre调整到cur之前了
            pre = cur;//pre 和 cur 后移一位
            cur = temp;
        }
        //开辟数组 尾插
        vector<int> ans;
        while(pre)
        {
            ans.push_back(pre->val);
            pre = pre->next;
        }
        return ans;
    }
};
```

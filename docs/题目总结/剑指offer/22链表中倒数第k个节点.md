```c
struct ListNode* getKthFromEnd(struct ListNode* head, int k){
    struct ListNode * cur = head;
    int num = 0;
    while(cur != NULL)//统计链表的长度
    {
        num++;
        cur = cur->next;
    }
    cur = head;
    while(num != k)
    {
        cur = cur->next;
        num--;
    }
    return cur;
}
```
